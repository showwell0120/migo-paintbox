# Button

> 按鈕。提供以下 props－

```javascript
type ColorType = 'default' | 'gray' | 'red';
type VariantType = 'default' | 'light' | 'outlined';

interface ReactButtonProps {
  color?: ColorType,          // 按鈕的顏色
  variant?: VariantType,      // 按鈕的形式
  disabled?: boolean,         // 使否為 disabled 的按鈕
  children?: React.ReactNode, // 按鈕上顯示的文字
  onClick?: () => void,       // 按鈕點擊時所觸法的 callback
}
```

## Usage

```jsx
import { ReactButton } from '@paintbox/react-button';

export const ButtonSample = () => {
  const onClick = () => {
    alert('You just clicked me!');
  };

  return (
    <div>
      <ReactButton variant={'light'} color={'gray'} onClick={onClick}>Click</ReactButton>
    </div>
  );
};
```
