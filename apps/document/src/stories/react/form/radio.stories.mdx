# Radio

> radio button, 提供以下 props －

```javascript
export interface ReactRadioProps {
  label: string,
  checked: boolean,
  name: string,
  value: string,
  onChange: (val: string) => void,
}
```

## Usage

```jsx
import { ReactRadio } from '@paintbox/react-radio';

export const ReactRadioSample = () => {
  const [check, setCheck] = React.useState('');

  function handleChange(val: string) {
    setCheck(val);
  }

  return (
    <ReactRadio label="test one" name="test" checked={check === 'test one'} value="test one" onChange={handleChange} />
  );
};
```
