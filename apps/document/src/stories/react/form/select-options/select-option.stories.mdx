# SelectOption

> 下拉選單。以控制元件的方式實作，提供以下 props－

```javascript
export interface ReactSelectOptionProps {
  options: OptionItem[];                   // 選項列表
  option?: OptionItem;                     // 目前選取的選項
  prefix?: string;                         // <select> 中的前綴字
  enableSearch?: boolean;                  // 是否允許符合字搜尋

  onChange: (item: OptionItem) => void;    // 變更選項的 event handler
}

export interface OptionItem {
  id: string | number;   // 選項 id
  name: string;          // 選項名稱
}
```

## Usage 

```jsx
import { ReactSelectOption, OptionItem } from '@paintbox/react-select-option';

const options: OptionItem[] = [
  {
    id: 'all',
    name: 'All',
  },
  {
    id: 'empty',
    name: 'Empty',
  },
];

export const SelectOptionSample = () => {
  const [selected, setSelected] = React.useState<boolean>();
  
  return <ReactSelectOption options={selected} option={selected} onChange={setSelected} />;
};
```
