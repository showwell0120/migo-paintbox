# Button

> 按鈕。提供以下 props－

```javascript
export interface ReactButtonProps {
  children: React.ReactNode;

  theme?: ColorThemeType;    // 從 react-foundation 匯入，按鈕的主題色
  variant?: Variant;         // 從 react-foundation 匯入，按鈕的形式
  disabled?: boolean;
  onClick?: () => void;
}
```

## Usage

```jsx
import { ReactButton } from '@paintbox/react-button';

export const ButtonSample = () => {
  const onClick = () => {
    alert('You just clicked me!');
  };

  return (
    <div>
      <ReactButton onClick={onClick}>Click me</ReactButton>
    </div>
  );
};
```
