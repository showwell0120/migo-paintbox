# SearchBar

> 搜索欄。以控制元件的方式實作，提供以下 props－

```javascript
export type CallbackProp = (t: string) => void

export interface ReactSearchBarProps {
  placeholder?: string,     // 搜索欄預設的顯示文字
  onChange?: CallbackProp,  // event handler for input change
  onEnter?: CallbackProp,   // event handler for press enter
  isDebounce?: boolean,     // 決定要不要使用 debounce, 只適用 onChange, 預設為 true
}
```

## Usage

```javascript
import { ReactSearchBar } from '@paintbox/react-search-bar';

export const Sample = () => {
    const fetchData = (keyword) => {
        // keyword is what you type in search bar
        // do what ever you want
    };

    return (
        <ReactSearchBar
            placeholder="default text"
            onChange={fetchData}
            onEnter={fetchData}
        />
    );
};

```
