# Toast

> Toast Message。提供以下 props－

```javascript
export type VariantType = 'warn' | 'info';

export interface ReactToastProps {
  variant: VariantType;    // toast 的形式
  children?: React.ReactNode;  // toast 內的顯示內容
  onClose?: (visible: boolean) => void;  // 關掉 toast 的 fallback
  yAxis?: 'top' | 'center' | 'bottom';   // toast 的 Y 軸位置
  xAxis?: 'right' | 'center' | 'left';   // toast 的 X 軸位置
  className?: string;    // 父元件定義的 css class 名稱
  waitToClose?: number;  // 自動關閉的秒數
  enableClose?: boolean; // 決定是否可以手動關閉 toast，預設為 true
}
```

## Usage

```jsx
import { ReactToast } from '@paintbox/react-toast';

export const ToastSample = () => {
  const [open, setOpen] = React.useState(false);

  const onClose = () => {
    setOpen(false);
  };

  return (
    <div>
      <button onClick={() => setOpen(true)}>Show message</button>
      {open && <ReactToast variant={'info'} onClose={onClose}>This is a reminder</ReactToast>}
    </div>
  );
};
```
