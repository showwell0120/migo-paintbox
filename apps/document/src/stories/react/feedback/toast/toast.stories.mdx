# Toast

> Toast Message。提供以下 props－

```javascript
export interface ReactToastProps {
  theme: Extract<
    ColorThemeType,
    | 'secondary-danger'
    | 'secondary-warning'
    | 'secondary-info'
    | 'secondary-success'
  >;
  children: React.ReactNode;
  yAxis?: 'top' | 'center' | 'bottom';    // 預設 top
  xAxis?: 'right' | 'center' | 'left';    // 預設 center
  className?: string;
  waitToClose?: number;                   // 多久之後關閉 toast
  enableClose?: boolean;
  showPrefixIcon?: boolean;               // 是否顯示前面的 icon

  onClose?: (visible: boolean) => void;
}
```

## Usage

```jsx
import { ReactToast } from '@paintbox/react-toast';

export const ToastSample = () => {
  const [open, setOpen] = React.useState(false);

  const onClose = () => {
    setOpen(false);
  };

  return (
    <div>
      <button onClick={() => setOpen(true)}>Show message</button>
      {open && <ReactToast theme={'secondary-danger'} onClose={onClose}>This is a reminder</ReactToast>}
    </div>
  );
};
```
