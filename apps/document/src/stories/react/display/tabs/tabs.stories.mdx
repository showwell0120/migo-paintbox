# Tabs

> 控制元件的形式。提供以下 props -

```javascript
export interface TabItem {
  id: number;
  name: string;
  count?: number;
}

/* eslint-disable-next-line */
export interface ReactTabsProps {
  tabs: TabItem[];                       // tab 選單資料
  currentTabID: number;                  // 目前選中的 tab ID
  onTabChange: (tabID: number) => void;  // 實作更新 tab 狀態的 callback
}
```

## Usage

```jsx
const Sample = () => {
  const [tabs, setTabs] = React.useState<TabItem[]>([
    { id: 1, name: 'All' },
    { id: 2, name: 'Video' },
    { id: 3, name: 'Dubbing' },
  ]);
  const [currentTabID, setCurrentTabID] = React.useState<number>(1);

  return (
    <ReactTabs
      tabs={tabs}
      currentTabID={currentTabID}
      onTabChange={setCurrentTabID}
    />
  );
};
```
