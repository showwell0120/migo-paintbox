# Chip

> Chip。提供以下 props - 

```javascript
type ColorType = 'green' | 'red' | 'blue' | 'white';
interface ColorSetting {
  text: string;
  background: string;
}

export interface ReactChipProps {
  color: ColorType;            // chip 的顏色
  children?: React.ReactNode;  // chip 內顯示的文字
  dot?: boolean;               // 是否要有左方的點點
  icon?: React.ReactNode;      // chip 內的 icon
  clickable?: boolean;         // 是否可以點選
  onClick?: () => void;        // 點選時所觸發的 function
}
```

## Usage
```jsx
import { ReactChip } from '@paintbox/react-chip';

export const ChipWithoutDotSample = () => {
  return (
    <div>
      <ReactChip
        color={'green'}
        dot={false}
        clickable={true}
      >Inactive</ReactChip>
    </div>
  );
};
```
