# Tag

> 標籤。提供以下 prop -

```javascript
type TextColorType = 'white' | 'default';
type BackgoundColorType = 'red' | 'blue' | 'purple' | 'default';

export interface ReactTagProps {
  text: string;                     // 標籤上的字
  colorType?: TextColorType;        // 標籤文字的顏色
  bgColorType?: BackgoundColorType; // 標籤的顏色
  hasBorder?: boolean;              // 是否要加外框
}
```

## Usage

```javascript
import { ReactTag } from '@paintbox/react-tag';

export const ColorTagSample = () => {
  return (
    <div>
      <ReactTag
        text={'test'}
        colorType={'white'}
        bgColorType={'blue'}
        hasBorder={true}
      />
    </div>
  );
};
```
