# StyledTable

> 表格相關組成的元件集合。需搭配 `react-table` 傳入各元件需要的 props。提供以下共同的 props－

```javascript
export interface TableBaseProps {
  children: React.ReactNode;
  className?: string;
}
```

## Usage 

```jsx
import { useTable,Column } from 'react-table'
import * as T from '@paintbox/react-table';

export const Sample = () => {
  const data = React.useMemo(
    () => [
      {
        col1: 'Hello',
        col2: 'World',
      },
      {
        col1: 'react-table',
        col2: 'rocks',
      },
      {
        col1: 'whatever',
        col2: 'you want',
      },
    ],
    []
  );

  const columns: ReadonlyArray<Column<{ col1: string; col2: string }>> =
    React.useMemo(
      () => [
        {
          Header: 'Column 1',
          accessor: 'col1', // accessor is the "key" in the data
        },
        {
          Header: 'Column 2',
          accessor: 'col2',
        },
      ],
      []
    );

  const tableInstance = useTable({ columns, data });

  const { getTableProps, getTableBodyProps, headerGroups, rows, prepareRow } =
    tableInstance;

  return (
    // apply the table props
    <T.StyledTable {...getTableProps()}>
      <T.StyledThead>
        {
          // Loop over the header rows
          headerGroups.map((headerGroup) => (
            // Apply the header row props
            <T.StyledHeadTR {...headerGroup.getHeaderGroupProps()}>
              {
                // Loop over the headers in each row
                headerGroup.headers.map((column) => (
                  // Apply the header cell props
                  <T.StyledTH {...column.getHeaderProps()}>
                    {
                      // Render the header
                      column.render('Header')
                    }
                  </T.StyledTH>
                ))
              }
            </T.StyledHeadTR>
          ))
        }
      </T.StyledThead>
      {/* Apply the table body props */}
      <T.StyledTbody {...getTableBodyProps()}>
        {
          // Loop over the table rows
          rows.map((row) => {
            // Prepare the row for display
            prepareRow(row);
            return (
              // Apply the row props
              <T.StyledBodyTR {...row.getRowProps()}>
                {
                  // Loop over the rows cells
                  row.cells.map((cell) => {
                    // Apply the cell props
                    return (
                      <T.StyledTD {...cell.getCellProps()}>
                        {
                          // Render the cell contents
                          cell.render('Cell')
                        }
                      </T.StyledTD>
                    );
                  })
                }
              </T.StyledBodyTR>
            );
          })
        }
      </T.StyledTbody>
    </T.StyledTable>
  );
};
```
